traffic_hourly <- traffic %>%
  mutate(
    date     = make_date(Yr, M, D),
    hour_num = as.integer(HH)
  ) %>%

pickup borough = PU_Borough

taxi_hourly <- taxi %>%
  mutate(
    date     = as.Date(date),
    hour_num = as.integer(substr(hour, 1, 2))
  ) %>%
  group_by(PU_Borough, date, hour_num) %>%
  summarise(
    trips              = sum(trip_count, na.rm = TRUE),
    avg_duration_min   = sum(duration_sum, na.rm = TRUE) / sum(trip_count, na.rm = TRUE),
    avg_distance_miles = sum(trip_distance_sum, na.rm = TRUE) / sum(trip_count, na.rm = TRUE),
    .groups = "drop"

# Add time_period to hourly_merged
hourly_period <- hourly_merged %>%
  mutate(
    time_period = case_when(
      hour_num >= 0  & hour_num <= 5  ~ "Overnight (0–5)",
      hour_num >= 7  & hour_num <= 9  ~ "AM Peak (7–9)",
      hour_num >= 11 & hour_num <= 14 ~ "Midday (11–14)",
      hour_num >= 16 & hour_num <= 19 ~ "PM Peak (16–19)",
      TRUE                            ~ "Other"
    )
  ) %>%
  # Only retain the few time periods we care about
  filter(time_period != "Other")

table1_period <- hourly_period %>%
  group_by(borough, time_period) %>%
  summarise(
    avg_traffic_volume     = mean(total_volume,     na.rm = TRUE),
    avg_trip_duration_min  = mean(avg_duration_min, na.rm = TRUE),
    avg_trip_distance_mile = mean(avg_distance_miles, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  arrange(borough, time_period)
